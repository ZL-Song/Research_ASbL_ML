## Read in data
# pathway = [ ( energy, progress ), ... ]
# pathways_list[ ( pathway, state, pathid ), ...]

basis_list = []
basis_names = ['sccdftb', '631G', '631+Gd', '631++Gdp', 'dftd631++Gdp', '6311++Gdp', 'dftd6311++Gdp']
for basis in basis_names:

    dat_in = open('../../6.data.prepare/2.ener.get/path_ener.{0}.dat'.format(basis), 'r')

    pathways_list = []

    # initialize parameters
    pathway = []
    state = 'none'
    pathid = 'nan'

    line = dat_in.readline()
    while line:
    
        if line == '===begin===\n':
            if len(pathway) > 1 and state != 'none' and pathid != 'nan':
                pathways_list.append(
                    (
                        state,
                        pathid,
                        pathway[:], 
                    )
                )
            pathway = []
            state = dat_in.readline().split().pop()
            pathid = dat_in.readline().split().pop()       # pathid line read
        elif line == '===+end+===\n':
            pathways_list.append(
                (
                    state,
                    pathid,
                    pathway[:], 
                )
            )
        else:
            words = line.split()
            pathway.append(
                (
                    float(words[1]),
                    int(words[0]),
                )
            )
        line = dat_in.readline()
    basis_list.append(pathways_list)
