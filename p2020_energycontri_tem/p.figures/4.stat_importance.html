## defs to load data
def build_dirs(basis, model_name, ):
    '''build the directores for all ds_ids to stat file 
    '''
    prefix = '../../9.fitting-man/2.testing'
    dir_list = [
        '{0}/{1}.ml_test/{2}/ds_{3}.stat'.format(
            prefix, basis, model_name, i, 
        ) for i in range(18)
    ]
    return dir_list

def load_stats(basis, model_name, which_feature):
    '''load all the stat files for a certain feature from all ds.
    '''
    dir_list = build_dirs(basis, model_name, )
    importance_list = []
    
    for d in dir_list:
        lines = open(d,'r').readlines()
        
        for l in lines:
            words = l.split()
            
            if words[0] == 'shff_rmse_{0}:'.format(which_feature):
                importance_list.append(float(words[3]))

    return importance_list
