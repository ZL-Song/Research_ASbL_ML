## defs to load data
def build_dir(basis, model_name, ds_id, is_man):
    '''build the directory to stat file.
    '''
    b_dir = '8.0.prediction-man' if is_man else '8.1.prediction'
    return '../../{0}/2.testing/{1}.ml_test/ds_{2}/{3}/stat'.format(b_dir, basis, ds_id, model_name, )

def get_stat(basis, model_name, ds_id, is_man):
    '''load data and return a tuple:
    
    (r2_score, average_error)
    '''
    # each file contains only 1 line.
    lines = open(build_dir(basis, model_name, ds_id, is_man), 'r').readlines()
    for l in lines:
        words = l.split()
        if words[0] == 'pred_rmse:':
            rmse = float(words[1])
        elif words[0] == 'r2_score:':
            r2 = float(words[1])
    return (r2, rmse)

def get_model_stat(basis, model_name, ds_ids, is_man):
    '''get all stat data of a model from ds_id in ds_ids
    ds_ids should be a list.
    
    returns:
    (
        [r2_score], [average_error]
    )
    '''
    dat = [ get_stat(basis, model_name, ds_id, is_man) for ds_id in ds_ids]
    return [r2[0] for r2 in dat], [ae[1] for ae in dat]

model_names=['svr', 'gpr', 'krr', ]
ds_ids=[i for i in range(0, 18)]

# colors
colors = [(230,  25,  75, ),
          ( 60, 180,  75, ), 
         ]
colors = [(float(r/255), float(g/255), float(b/255),) for r,g,b in colors]


# props of box plot elements
capprops = dict(linewidth=0.5, color=colors[0])
boxprops = dict(linewidth=0.5, color=colors[0])
whiskerprops = dict(linewidth=0.5, color=colors[0])
flierprops = dict(marker='.', markerfacecolor=colors[0], markeredgecolor=colors[0], markersize=2, )
medianprops = dict(linewidth=0.5, linestyle='-', color='k')

man_capprops = dict(linewidth=0.5, color=colors[1])
man_boxprops = dict(linewidth=0.5, color=colors[1])
man_whiskerprops = dict(linewidth=0.5, color=colors[1])
man_flierprops = dict(marker='.', markerfacecolor=colors[1], markeredgecolor=colors[1], markersize=2, )
man_medianprops = dict(linewidth=0.5, linestyle='-', color='k')

#fig setup
fig, axes = plt.subplots(nrows=3, 
                         sharey='row', 
                         sharex='col', 
                         figsize=(4, 12), 
                         dpi=300, 
                        )

# width of each box
boxwidth = 0.5

for i in range(len(axes)):
    ax=axes[i]
    
    # y=2 line
    ax.plot([-1, 18.0], [2.0, 2.0], c = 'gray',ls='-', linewidth=0.5)
    
    ## plot box
    # get data
    m = 'svr' if i == 0 else 'gpr' if i == 1 else 'krr'
    rfe_dat = [
        get_model_stat(basis, m, ds_ids, False) for basis in basis_list
    ]
    man_dat = [
        get_model_stat(basis, m, ds_ids, True) for basis in basis_list
    ]
    
    # plot manual feature selection result
    mae_man = ax.boxplot(
        [d[1] for d in man_dat],           # list of average errors
        vert=True,
        notch=False, 
        patch_artist=True, 
        positions=[0.5+x*2.5 for x in range(7)],            # len(labels)
        capprops=man_capprops,
        boxprops=man_boxprops,
        whiskerprops=man_whiskerprops,
        flierprops=man_flierprops,
        medianprops=man_medianprops,
        widths=(boxwidth, boxwidth, boxwidth, boxwidth, boxwidth, boxwidth, boxwidth, ),
    )
    for patch in mae_man['boxes']:
        patch.set(facecolor=colors[1])
        patch.set(edgecolor=colors[1])
    
    # plot manual feature selection result
    mae = ax.boxplot(
        [d[1] for d in rfe_dat],           # list of average errors
        vert=True,
        notch=False,
        patch_artist=True, 
        positions=[1.5+x*2.5 for x in range(7)],            # len(labels)
        capprops=capprops,
        boxprops=boxprops,
        whiskerprops=whiskerprops,
        flierprops=flierprops,
        medianprops=medianprops,
        widths=(boxwidth, boxwidth, boxwidth, boxwidth, boxwidth, boxwidth, boxwidth, ),
    )
    for patch in mae['boxes']:
        patch.set(facecolor=colors[0])
        patch.set(edgecolor=colors[0])
        
    # spine
