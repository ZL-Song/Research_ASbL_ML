## defs to load data
def build_dir(basis, model_name, ds_id, ):
    '''build the directory to stat file.
    '''
    return '../../8.2.converg-test/2.testing/{0}.ml_test/{2}_test/size{1}_stat'.format(
        basis, ds_id, model_name, 
    )

def get_stat(basis, model_name, ds_id, ):
    '''load data and return a tuple:
    
    (r2_score, average_error)
    '''
    # each file contains only 1 line.
    lines = open(build_dir(basis, model_name, ds_id, ), 'r').readlines()
    for l in lines:
        words = l.split()
        if words[0] == 'pred_rmse:':
            rmse = float(words[1])
        elif words[0] == 'r2_score:':
            r2 = float(words[1])
    return (r2, rmse)

def get_model_stat(basis, model_name, ds_ids, ):
    '''get all stat data of a model from ds_id in ds_ids
    ds_ids should be a list.
    
    returns:
    (
        [r2_score], [average_error]
    )
    '''
    dat = [ get_stat(basis, model_name, ds_id, ) for ds_id in ds_ids]
    return [r2[0] for r2 in dat], [ae[1] for ae in dat]
